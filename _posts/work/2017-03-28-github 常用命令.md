```c++

1.git clone "ssh"; // 克隆远程仓库
 git clone "ssh" home-page; // 克隆到指定仓库

2.若使用文件夹，手动把文件夹内文件移至仓库根目录

3.git status; // 检测本地文件状态（红，未跟踪）

4.git add --all; // 克隆远程仓库

5.git status; // 再检测

6.git commit -m 'initial'; // 本次操作注释

7.git push -u origin master; // 提交到远程仓库 （u，初次使用）



```
