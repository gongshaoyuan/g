```c++

ctrl + n 	// 新建文件
html:xt 	// 回车，html骨架
html:5 	// html5骨架
ctrl + tab 	// 切换标签页面
ctrl + g 	// 跳转至第几行
ctrl + w 	// 关闭当前文件
ctrl + l 	// 选择行，重复可选择多行
ctrl + x 	// 删除当前行
ctrl + h 	// 替换
ctrl + F2 	// 设置/删除标记

ctrl + d 	// 选中光标所占文本，重复选择相同
ctrl + F3 	// 先选中文本按下，选择全部相同
ctrl 	// 按住点击可自定义选择多处
ctrl + l 	// 选择整行
shift + tab 	// 往回缩进
home 	// 光标移至行开头
end 	// 光标移至行结尾

ctrl + / 	// 注释
ctrl + alt + f 	// 格式化


ctrl + shift + d 	// 复制当前行
ctrl + shift + k 	// 删除当前行
ctrl + shift + 上、下箭头 	// 上移下移当前行
ctrl + shift + [ 	// 折叠
ctrl + shift + ] 	// 展开
ctrl + shift + p 	// 打开命令面板

```
